414                            winners(win_unit,2) = winners(win_unit,2) + p.nRows;

416                    end

419                    %    act_out(isnan(act_out))=0;
420                    % end
421

546                    %         p.meanSelectivity_caudal_new(trial,:) = judging.initial_selec(layer_new,:).*featuresToCompare; % only look at layers for which features were sampled
547                    meanSelectivity_caudal_new = judging.initial_selec(layer_new,:).*featuresToCompare;
548                    
549                    
550                elseif layer_new == 2 && (sum(p.usePRC(:,trial))==2)
551

555                    
556

563                if layer_prev == 1

565                    meanSelectivity_caudal_prev = judging.selectivity_prev(layer_prev,:).*featuresToCompare;
566                    
567                    familDiff_caudal = abs(meanSelectivity_caudal_prev - meanSelectivity_caudal_new);
568                    
569                    % determine which grid in caudal layer would be used for judgement
570                    whichCaudal = find(familDiff_caudal==max(familDiff_caudal));
571                    
572                    %                 if length(whichCaudal)>1
573                    whichCaudal = whichCaudal(1);
574                    %                 end
575                    
576                    if isempty(whichCaudal)
577                        2;
578                    end
579                    
580                    p.meanSelectivity_caudal_prev(trial) = meanSelectivity_caudal_prev(whichCaudal);
581                    p.meanSelectivity_caudal_new(trial) = meanSelectivity_caudal_new(whichCaudal);
582                    
583                    p.familDiff_caudal(trial) = familDiff_caudal(whichCaudal);
584                    
585                elseif (layer_prev == 2) && (all(p.usePRC(:,trial)))
586                    p.meanSelectivity_PRC_prev(trial) = judging.selectivity_prev(layer_prev,1);
587                    p.familDiff_PRC(trial) = abs(p.meanSelectivity_PRC_prev(trial) - p.meanSelectivity_PRC_new(trial));
588                    
589                end
590            end
591            
592            
593            if ~p.diffEncode
594                
595                % only use PRC layer if p.numGrids_Caudal features were sampled for both stimuli
596                if (all(p.usePRC(:,trial))) && (p.which_gp_layer==2)
597                    familDiff = p.familDiff_PRC(trial);
598                    thresh = p.famil_diff_thresh(p.numThresh,:);
599                else
600                    familDiff = p.familDiff_caudal(trial);
601                    thresh = p.famil_diff_thresh(1,:);
602                end
603                
604            else
605                
606                if p.layer == 2
607                    
608                    familDiffs_temp = [p.familDiff_caudal(trial) - mean(p.famil_diff_thresh(1,:)), p.familDiff_PRC(trial) - mean(p.famil_diff_thresh(p.numThresh,:))];
609                    [~, whichFamilDiff] = max(familDiffs_temp);
610                    
611                    if whichFamilDiff == 2
612                        familDiff = p.familDiff_PRC(trial);
613                        thresh = p.famil_diff_thresh(p.numThresh,:);
614                    else
615                        familDiff = p.familDiff_caudal(trial);
616                        thresh = p.famil_diff_thresh(1,:);
617                        p.usePRC(:,trial) = 0;
618                    end
619                    
620                else
621                    familDiff = p.familDiff_caudal(trial);
622                    thresh = p.famil_diff_thresh(1,:);
623                    p.usePRC(:,trial) = 0;
624                    
625                end
626                
627            end
628            %% compare differences in selectivity with threshold
629            % p.famil_difference_most(trial) = abs(mostSelectiveLayer_new - mostSelectiveLayer_prev);% /(mostSelectiveLayer_new+mostSelectiveLayer_prev); %+ (rand*2-1)*p.decision_noise;

633            p.familDiff_withNoise(trial) = (familDiff-noise)+(familDiff-(familDiff-noise)).*rand; %familDiff + noise;%  ((rand(1,5)*2-1)*p.decision_noise);
634            % p.familDiff_withNoise(trial) = familDiff;
635            % thresh = (thresh-noise)+(thresh-(thresh-noise)).*rand;
636            
637            if any(p.familDiff_withNoise(trial) > (mean(thresh)))
638                % we have misMatching feature! So, stop sampling

640                
641            else
642                % no evidence for misMatch of stimuli
643                stopSampling = 0;
644                
645            end
646            
647            % temporarily commenting out to see if caudal layre can be same in lesion
648            % and control
649

651            % if (p.layer == 2) && (all(p.usePRC(:,trial))) % if PRC layer not available, or not enough features were sampled...
652            %     p.famil_difference(trial) = familDiff;
653            % else
654            %     %     because sometimes two grids can match, need to select just one of
655            %     %     them
656            %     famil_diff_temp = familDiff(whichCaudal);
657            %     p.famil_difference(trial) = famil_diff_temp(1);
658            % end
659

661

663

669            %------------------------------------------------------------------

671                continue;
672            end

682        prevStimSelec = selectivity;
683        %     prevInitialSelec = initial_selec;
684        %     prevInitialActs = initial_acts;
685        %     prev_mean_act = mean_act;
686       
687        
688        %     prevStimActs = acts;
689

691        p.usePRC(stim,trial) = usePRC;
692        
693        
694        %----------------------------------------------------------------------
695        %switch to other stim (last thing you do)
696        %----------------------------------------------------------------------
697        
698        stim = (stim+2)/stim - 1; %1 goes to 2, 2 goes to 1
699        if first_stim_sampled == 1 && p.fixations(trial) >= p.maxFix
700            keepSampling = 1;
701        elseif first_stim_sampled == 0
702            keepSampling = 0;
703        end
704        first_stim_sampled = 0; %to indicate that we are no longer on the first stim being sampled (so a comparison should be made from now on)
705
